#include <iostream>
#include <GL/glut.h>
using namespace std;
int v, choice;
int xSeed, ySeed;
float fillColor[3];
float boundaryColor[3] = {0.0, 0.0, 0.0};
float bgColor[3] = {1.0, 1.0, 1.0};
int input[2][20];
int Round(float a) {
    return (a + 0.5);
}

void getPixel(int x, int y, float color[3]) {
    glReadPixels(x, y, 1, 1, GL_RGB, GL_FLOAT, color);
}

void setPixel(int x, int y, float color[3]) {
    glColor3f(color[0], color[1], color[2]);
    glBegin(GL_POINTS);
    glVertex2i(x, y);
    glEnd();
    glFlush();
}
void drawLine(int x1, int y1, int x2, int y2) {
    int dx = x2 - x1;
    int dy = y2 - y1;

   int steps = abs(dx) > abs(dy) ? abs(dx) : abs(dy);

    float xinc = dx / (float)steps;
    float yinc = dy / (float)steps;
    float x = x1, y = y1;
    glBegin(GL_POINTS);
    for (int i = 0; i <= steps; i++) {
        glVertex2i(Round(x), Round(y));
        x += xinc;
        y += yinc;
    }
    glEnd();
}
void drawPolygon() {
    glColor3f(boundaryColor[0], boundaryColor[1], boundaryColor[2]);

    for (int i = 0; i < v; i++) {
        drawLine(input[0][i], input[1][i],
                 input[0][(i + 1) % v], input[1][(i + 1) % v]);
    }
}
void boundaryFill(int x, int y) {
    float color[3];
    getPixel(x, y, color);

    if ((color[0] != boundaryColor[0] || color[1] != boundaryColor[1] || color[2] != boundaryColor[2]) &&
        (color[0] != fillColor[0] || color[1] != fillColor[1] || color[2] != fillColor[2])) {

        setPixel(x, y, fillColor);

        boundaryFill(x + 1, y);
        boundaryFill(x - 1, y);
        boundaryFill(x, y + 1);
        boundaryFill(x, y - 1);
    }
}
void floodFill(int x, int y) {
    float color[3];
    getPixel(x, y, color);

    if ((color[0] == bgColor[0] && color[1] == bgColor[1] && color[2] == bgColor[2]) &&
        (color[0] != fillColor[0] || color[1] != fillColor[1] || color[2] != fillColor[2])) {
        setPixel(x, y, fillColor);
        floodFill(x + 1, y);
        floodFill(x - 1, y);
        floodFill(x, y + 1);
        floodFill(x, y - 1);
    }
}
void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    drawPolygon();

    switch (choice) {
        case 1:
            boundaryFill(xSeed, ySeed);
            break;
        case 2:
            floodFill(xSeed, ySeed);
            break;
        default:
            cout << "Invalid choice!" << endl;
    }

    glFlush();
}
int main(int argc, char** argv) {
    cout << "\nChoose Algorithm:\n";
    cout << "1. Boundary Fill\n";
    cout << "2. Flood Fill\n";
    cout << "Enter choice: ";
    cin >> choice;

    cout << "\nEnter number of vertices: ";
    cin >> v;

    for (int i = 0; i < v; i++) {
        cout << "Enter x and y: ";
        cin >> input[0][i] >> input[1][i];
    }

    cout << "\nEnter seed point (inside polygon): ";
    cin >> xSeed >> ySeed;

    cout << "\nEnter fill color (RGB in range 0-1): ";
    cin >> fillColor[0] >> fillColor[1] >> fillColor[2];


    if (choice == 1) {
        cout << "\nEnter boundary color (RGB in range 0-1): ";
        cin >> boundaryColor[0] >> boundaryColor[1] >> boundaryColor[2];
    }
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(600, 600);
    glutCreateWindow("Boundary Fill & Flood Fill");
    glClearColor(bgColor[0], bgColor[1], bgColor[2], 1.0);
    gluOrtho2D(0, 600, 0, 600);
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
