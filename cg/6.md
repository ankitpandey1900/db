#include<iostream>
#include<math.h>
#include<GL/glut.h>
using namespace std;

int v,frx,fry,fsx,fsy,choice;
int input[3][10];
float output[3][10];
float tx,ty,sx,sy,ang,r;

void multiplication(int input[3][10], float output[3][10], float f[3][3])
{
    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < v; j++)
        {
            output[i][j] = 0;
            for(int k = 0; k < 3; k++)
            {
                output[i][j] += f[i][k] * input[k][j];
            }
        }
    }

    for(int a = 0; a < 3; a++)
    {
        for(int b = 0; b < v; b++)
        {
            cout << output[a][b] << "\t";
        }
        cout << "\n";
    }
}

void translation()
{
    float f[3][3] = {
                 {1, 0, tx},
                     {0, 1, ty},
                     {0, 0, 1}};

    multiplication(input, output, f);
}

void rotation()
{   
    r = M_PI * ang / 180;
    float f[3][3] = {
                 {cos(r), -sin(r), frx*(1-cos(r))+(fry*sin(r))},
                     {sin(r), cos(r), fry*(1-cos(r))-(frx*sin(r))},
                     {0, 0, 1}};

    multiplication(input, output, f);
}

void scaling()
{
    float f[3][3] = {
                 {sx, 0, fsx*(1-sx)},
                     {0, sy, fsy*(1-sy)},
                     {0, 0, 1}};

    multiplication(input, output, f);
}

void Display()
{
    glClear(GL_COLOR_BUFFER_BIT);
    
    glColor3f(0,0,0);
    glBegin(GL_LINE_LOOP);
    for(int i = 0; i<v; i++){
        glVertex2f(input[0][i],input[1][i]);
    }
        glEnd();
    
    glColor3f(0,0,1);
    glBegin(GL_LINE_LOOP);
        for(int i = 0; i<v; i++){
        glVertex2f(output[0][i],output[1][i]);
    }
        glEnd();
glFlush();
}

int main(int argc, char**argv)
{
    cout << "Enter the Total Number of Vertices: ";
    cin >> v;

    cout << "Enter all the vertices: " << endl;
    for(int i = 0; i < v; i++)
    {
        cout << "Enter x value : ";
        cin >> input[0][i];
        cout << "Enter y value : ";
        cin >> input[1][i];
        input[2][i] = 1;
    }

    do
    {
        cout << "Choose the desired Operation" << endl;
        cout << "1. Translation" << endl;
        cout << "2. Rotation" << endl;
        cout << "3. Scaling" << endl;
        cout << "4. Display" << endl;
        cout << "What do you want to perform? :";
        cin >> choice;

        if(choice == 1)
        {
            cout << "Enter the Translating Factors" << endl;
            cout << "Tx: ";
            cin >> tx;
            cout << "Ty: ";
            cin >> ty;
            translation();
        }
        else if(choice == 2)
        {
            cout << "Enter the Angle of Rotation (Anticlockwise)" << endl;
            cout << "Angle: ";
            cin >> ang;
            cout << "Enter the Points of Rotation" << endl;
            cout << "Frx: ";
            cin >> frx;
            cout << "Fry: ";
            cin >> fry;
        }
        else if(choice == 3)
        {
            cout << "Enter the Scaling Factors" << endl;
            cout << "Sx: ";
            cin >> sx;
            cout << "Sy: ";
            cin >> sy;
            cout << "Enter the Points of Scaling" << endl;
            cout << "Fsx: ";
            cin >> fsx;
            cout << "Fsy: ";
            cin >> fsy;
        }
        else if(choice == 4)
        {
            glutInit(&argc, argv);
        glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
        glutInitWindowSize(600,600);
        glutInitWindowPosition(100,100);
        glutCreateWindow("2D");
        glClearColor(1,1,1,1);
        gluOrtho2D(0,640,0,480);
        glutDisplayFunc(Display);
        glutMainLoop();
            
        }

    } while(choice <= 3);
    
    return 0;
}
