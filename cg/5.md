#include <iostream>
#include <GL/glut.h>
using namespace std;
int xmin, ymin, xmax, ymax;
int v;
int input[2][20];
int output[2][20];
int k = 0;
void left_clip(int xa, int ya, int xb, int yb)
{
    float m = (float)(yb - ya) / (xb - xa);
    if (xa >= xmin && xb >= xmin) // in-in
    {
        output[0][k] = xb;
        output[1][k] = yb;
        k++;
    }
    else if (xa < xmin && xb >= xmin) // out-in
    {
        output[0][k] = xmin;
        output[1][k] = ya + m * (xmin - xa);
        k++;
        output[0][k] = xb;
        output[1][k] = yb;
        k++;
    }
    else if (xa >= xmin && xb < xmin) // in-out
    {
        output[0][k] = xmin;
        output[1][k] = ya + m * (xmin - xa);
        k++;
    }
}

void display()
{
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(0, 0, 1);
    glBegin(GL_LINE_LOOP);
