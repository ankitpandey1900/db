    }

        outcode2[3] = 1;

        outcode2[1] = 1;

    float dy = y2 - y1;
    float m = dy / dx;

    {
        if (outcode1[i] == 1 || outcode2[i] == 1)
        {
            flag = 1;
            break;
        }
