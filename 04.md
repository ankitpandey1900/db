## 1) Products per category

```sql
SELECT c.category_name, COUNT(p.product_id) AS total_products
FROM categories c
LEFT JOIN products p ON c.category_id = p.category_id
GROUP BY c.category_name;
```

## 4) View: low stock (<10 total)

```sql
CREATE VIEW low_stock AS
SELECT p.product_id, p.name, SUM(i.quantity_in_stock) AS total_stock
FROM products p
JOIN inventory i ON p.product_id = i.product_id
GROUP BY p.product_id, p.name
HAVING SUM(i.quantity_in_stock) < 10;
```

## 6) Trigger: prevent negative stock

```sql
DELIMITER //

CREATE TRIGGER no_negative_stock
BEFORE INSERT ON inventory
FOR EACH ROW
BEGIN
    IF NEW.quantity_in_stock < 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Invalid stock';
    END IF;
END //

DELIMITER ;
```

## 4) Cursor: display product stock

```sql
DELIMITER //

CREATE PROCEDURE product_stock_cursor()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE pid INT;
    DECLARE total INT;

    DECLARE cur CURSOR FOR
    SELECT product_id, SUM(quantity_in_stock)
    FROM inventory
    GROUP BY product_id;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;

    OPEN cur;

    loop1: LOOP
        FETCH cur INTO pid, total;
        IF done THEN LEAVE loop1; END IF;
        SELECT pid, total;
    END LOOP;

    CLOSE cur;
END //

DELIMITER ;
```
