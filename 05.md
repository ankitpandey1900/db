## 6) Cursor: patients >60

```sql id="n5qlap"
DELIMITER //

CREATE PROCEDURE senior_patients()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE pname VARCHAR(50);
    DECLARE phone VARCHAR(15);

    DECLARE cur CURSOR FOR
    SELECT patient_name, phoneNumber
    FROM patients
    WHERE TIMESTAMPDIFF(YEAR, DOB, CURDATE()) > 60;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;

    OPEN cur;

    loop1: LOOP
        FETCH cur INTO pname, phone;
        IF done THEN LEAVE loop1; END IF;
        SELECT pname, phone;
    END LOOP;

    CLOSE cur;
END //

DELIMITER ;
```

    SELECT 
        SUM(CASE WHEN age < 18 THEN 1 ELSE 0 END) AS minors,
        SUM(CASE WHEN age BETWEEN 18 AND 60 THEN 1 ELSE 0 END) AS adults,
        SUM(CASE WHEN age > 60 THEN 1 ELSE 0 END) AS seniors
