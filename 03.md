## 1) Total sales per category

```sql
SELECT m.category, SUM(oi.quantity * oi.unit_price) AS total_sales
FROM menu m
JOIN orderitems oi ON m.item_id = oi.item_id
GROUP BY m.category;
```

### 🧠 Why

* Revenue = quantity × price
* Group by category

### 🧠 HinEnglish

"Har item ka paisa nikala → category wise sum."

## 2) Procedure: order details by ID

```sql
DELIMITER //

CREATE PROCEDURE get_order_details(IN oid INT)
BEGIN
    SELECT *
    FROM orders o
    JOIN orderitems oi ON o.order_id = oi.order_id
    WHERE o.order_id = oid;
END //

DELIMITER ;
```

## 5) View: top 3 selling items

```sql
CREATE VIEW top_items AS
SELECT m.item_name, SUM(oi.quantity) AS total_sold
FROM menu m
JOIN orderitems oi ON m.item_id = oi.item_id
GROUP BY m.item_name
ORDER BY total_sold DESC
LIMIT 3;
```

## 7) Cursor: daily sales summary

```sql
DELIMITER //

CREATE PROCEDURE daily_sales()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE total DECIMAL(10,2);

    DECLARE cur CURSOR FOR
    SELECT SUM(quantity * unit_price) FROM orderitems;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;

    OPEN cur;

    read_loop: LOOP
        FETCH cur INTO total;
        IF done THEN LEAVE read_loop; END IF;
        SELECT total;
    END LOOP;

    CLOSE cur;
END //

DELIMITER ;
```

```sql
DELIMITER //

CREATE FUNCTION check_category(cat VARCHAR(50))
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE cnt INT;
    SELECT COUNT(*) INTO cnt FROM menu WHERE category = cat;
    RETURN cnt;
END //

DELIMITER ;
```

## 7) Trigger: prevent negative price

```sql
DELIMITER //

CREATE TRIGGER no_negative_price
BEFORE INSERT ON menu
FOR EACH ROW
BEGIN
    IF NEW.price < 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Invalid price';
    END IF;
END //

DELIMITER ;
```
